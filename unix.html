//by Han-Meng

//Unix常用指令

/**
 //以下是Apache操作相关
 # 关闭 apache 服务器
 $ sudo apachectl -k stop

 # 启动 apache 服务器
 $ sudo apachectl -k start

 # 重新再次启动 apache
 $ sudo apachectl -k restart

 //以下是文件操作相关
 cd 切换目录
 pwd 确认当前目录
 ls 列表显示当前目录的内容
 cat 查看文件,包括能查看二进制文件
 touch 新建立一个文件
 open 打开一个文件
 sudo 表示需要管理员权限
 echo 后面一般接字符串

 //以下是BASE64相关
 base64 对二进制文件进行编码成65位长度字符串
 -o 输出成xx文件
 -D 解码

 # 将字符串 Man 进行 base64 编码
 $ echo -n "Man" | base64

 # 将字符串 TWFu 解码
 $ echo -n "TWFu" | base64 -D


 //以下是MD5相关
 # 得到文件的MD5值
 $ md5 文件名

 # 得到字符串的MD5值
 md5 -s "string"

 // 查看本机ip地址
 ifconfig

 **/
